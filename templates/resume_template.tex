% --- Custom Section Command ---
% Creates a clean section title with a horizontal rule below it.
\newcommand{\resumesection}[1]{
  \vspace{8pt} % Space above the section title
  {\large\bfseries\scshape\color{black}#1} % Section title text
  \vspace{3pt} % Space between title and rule
  \hrule height 0.8pt % A slightly thicker, more prominent rule
  \vspace{8pt} % Space below the rule
}
